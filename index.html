# pip install alpaca-trade-api requests yfinance pandas # # Day-trading OPTIONS bot — buys calls on bullish signals, puts on bearish # signals, using the same 5-min MA/EMA/volume signal engine as the stock bot. # Every position is closed by end of day — this bot never holds overnight. # # Runs as its own set of Render cron jobs, separate from trading_bot.py: # - options_bot.py scan -> every 5 min, market hours # - options_bot.py manage -> every 1 min, market hours (stop/target/EOD flatten) # # Uses the SAME Alpaca account/keys as your stock bot — both draw on the same # buying power. Keep that in mind when sizing trades across both bots. import datetime import os import sys import time import zoneinfo import alpaca_trade_api as tradeapi import pandas as pd import requests import yfinance as yf # ============================================================================== # STEP 1: CONFIGURATION & CREDENTIALS # ============================================================================== REQUIRED_VARS = ["ALPACA_API_KEY", "ALPACA_SECRET_KEY", "ALPACA_BASE_URL", "DISCORD_WEBHOOK_URL"] def check_required_env_vars(): missing = [name for name in REQUIRED_VARS if not os.environ.get(name, "").strip()] print("🔎 Credential check:") for name in REQUIRED_VARS: value = os.environ.get(name) print(f" {name}: {'present (length ' + str(len(value)) + ')' if value else '❌ MISSING or empty'}") if missing: print("\n❌ STARTUP FAILED — missing: " + ", ".join(missing)) sys.exit(1) check_required_env_vars() ALPACA_API_KEY = os.environ["ALPACA_API_KEY"] ALPACA_SECRET_KEY = os.environ["ALPACA_SECRET_KEY"] ALPACA_BASE_URL = os.environ["ALPACA_BASE_URL"] DISCORD_WEBHOOK_URL = os.environ["DISCORD_WEBHOOK_URL"] print(f"🔗 Alpaca base URL: {ALPACA_BASE_URL}") if "paper" in ALPACA_BASE_URL: print("⚠️ PAPER trading mode active. No real money at risk.") else: print("🟣 LIVE trading mode active. REAL money is at risk.") # Same watchlist as the stock bot — all liquid, listed-options names WATCHLIST = [ "AAPL", "MSFT", "GOOGL", "NVDA", "SPY", "AMZN", "JNJ", "UNH", "JPM", "BAC", "XOM", "CVX", "WMT", "COST", "QQQ", "IWM", ] # --- Signal engine (same as stock bot: 5-min bars, 20-bar MA, 200 EMA trend filter, volume surge) --- BAR_INTERVAL = "5m" LOOKBACK_PERIOD = "30d" MA_WINDOW = 20 EMA_TREND_WINDOW = 200 VOLUME_AVG_WINDOW = 20 VOLUME_SURGE_MULTIPLIER = 1.5 RECENT_ENTRY_BLOCK_MINUTES = 5 # --- Options-specific contract selection --- TARGET_EXPIRATION_MIN_DAYS = 0 TARGET_EXPIRATION_MAX_DAYS = 2 # --- Options-specific sizing & risk (premium-based, NOT stock-price-based) --- BUDGET_PER_TRADE = 100.0 # target $ spend on premium per trade MAX_CONTRACT_COST_CAP = 300.0 # skip the trade if even 1 contract costs more than this OPTION_STOP_LOSS_PCT = 0.30 # exit if premium drops 30% from entry OPTION_TAKE_PROFIT_PCT = 0.50 # exit if premium gains 50% from entry (~1.67:1 reward:risk) # --- Day-trading rule: never hold overnight --- EOD_FLATTEN_MINUTES_BEFORE_CLOSE = 15 TRADING_BASE = ALPACA_BASE_URL.rstrip("/") DATA_BASE = "https://data.alpaca.markets" HEADERS = {"APCA-API-KEY-ID": ALPACA_API_KEY, "APCA-API-SECRET-KEY": ALPACA_SECRET_KEY} api = tradeapi.REST(ALPACA_API_KEY, ALPACA_SECRET_KEY, ALPACA_BASE_URL, api_version="v2") # ============================================================================== # STEP 2: NOTIFICATIONS (Discord) # ============================================================================== def send_alert(message_text): try: for i in range(0, len(message_text), 2000): chunk = message_text[i:i + 2000] response = requests.post(DISCORD_WEBHOOK_URL, json={"content": chunk}, timeout=10) if response.status_code != 204: print(f"❌ Discord Notification failed: {response.status_code} {response.text}") except Exception as e: print(f"❌ Notification connection error: {e}") # ============================================================================== # STEP 3: OPTION CONTRACT DISCOVERY & PRICING # ============================================================================== def get_option_contracts(underlying, option_type): """Fetch active contracts for `underlying` (call/put) expiring within the target window, via Alpaca's Trading API options-contracts endpoint.""" today = datetime.date.today() exp_gte = (today + datetime.timedelta(days=TARGET_EXPIRATION_MIN_DAYS)).isoformat() exp_lte = (today + datetime.timedelta(days=TARGET_EXPIRATION_MAX_DAYS)).isoformat() try: resp = requests.get( f"{TRADING_BASE}/v2/options/contracts", headers=HEADERS, params={ "underlying_symbols": underlying, "expiration_date_gte": exp_gte, "expiration_date_lte": exp_lte, "type": option_type, "status": "active", "limit": 100, }, timeout=15, ) resp.raise_for_status() return resp.json().get("option_contracts", []) except Exception as e: print(f"⚠️ Could not fetch option contracts for {underlying} ({option_type}): {e}") return [] def choose_atm_contract(contracts, current_price): """Pick the contract with strike closest to the current stock price (at-the-money), preferring the expiration closest to the middle of the target window if there's a tie.""" if not contracts: return None target_days = (TARGET_EXPIRATION_MIN_DAYS + TARGET_EXPIRATION_MAX_DAYS) / 2 today = datetime.date.today() def score(c): strike_diff = abs(float(c["strike_price"]) - current_price) exp_date = datetime.date.fromisoformat(c["expiration_date"]) days_diff = abs((exp_date - today).days - target_days) return (strike_diff, days_diff) return min(contracts, key=score) def get_option_mid_price(option_symbol): """Latest bid/ask midpoint for an option contract, via Alpaca's options market data endpoint (indicative feed — free tier, available on paper).""" try: resp = requests.get( f"{DATA_BASE}/v1beta1/options/quotes/latest", headers=HEADERS, params={"symbols": option_symbol, "feed": "indicative"}, timeout=15, ) resp.raise_for_status() quote = resp.json().get("quotes", {}).get(option_symbol) if not quote: return None bid = float(quote.get("bp", 0) or 0) ask = float(quote.get("ap", 0) or 0) if bid <= 0 or ask <= 0: return None return (bid + ask) / 2 except Exception as e: print(f"⚠️ Could not fetch quote for {option_symbol}: {e}") return None # ============================================================================== # STEP 4: ORDER SUBMISSION (same Orders API as stock trades) # ============================================================================== def submit_option_order_and_notify(option_symbol, qty, side, reason): try: order = api.submit_order( symbol=option_symbol, qty=qty, side=side, type="market", time_in_force="day" ) except Exception as e: send_alert(f"❌ OPTION ORDER FAILED:\nAction: {side.upper()} {qty}x {option_symbol}\nError: {e}") print(f"Status: Option order submission failed for {option_symbol}: {e}") return final_order = order for _ in range(5): time.sleep(2) try: final_order = api.get_order(order.id) except Exception: break if final_order.status in ("filled", "canceled", "rejected", "expired"): break status = final_order.status if status == "filled": fill_price = final_order.filled_avg_price price_str = f"${float(fill_price):.2f}" if fill_price else "market price" send_alert( f"✅ OPTION ORDER FILLED:\n" f"💰 {side.upper()} {qty}x {option_symbol}\n" f"📈 Fill Price: {price_str}/contract\n" f"🟢 Reason: {reason}" ) print(f"Status: {side.upper()} {option_symbol} filled at {price_str}.") elif status in ("canceled", "rejected", "expired"): send_alert( f"🚫 OPTION ORDER {status.upper()}:\n" f"💰 {side.upper()} {qty}x {option_symbol}\n" f"🔴 Reason attempted: {reason}" ) else: send_alert(f"⏳ OPTION ORDER STATUS UNCLEAR: {side.upper()} {qty}x {option_symbol} — verify manually.") # ============================================================================== # STEP 5: SIGNAL ENGINE (identical logic to the stock bot's scan) # ============================================================================== def get_signal(ticker): """Returns 'bullish', 'bearish', or None, using the same 20-bar MA / 200 EMA trend / volume-surge rule as trading_bot.py.""" data = yf.download(ticker, period=LOOKBACK_PERIOD, interval=BAR_INTERVAL, progress=False) if data.empty: return None, None if isinstance(data.columns, pd.MultiIndex): data.columns = data.columns.get_level_values(0) data["MA"] = data["Close"].rolling(window=MA_WINDOW).mean() data["EMA200"] = data["Close"].ewm(span=EMA_TREND_WINDOW, adjust=False).mean() data["VolAvg"] = data["Volume"].rolling(window=VOLUME_AVG_WINDOW).mean() min_bars_needed = max(MA_WINDOW, EMA_TREND_WINDOW, VOLUME_AVG_WINDOW) + 1 if len(data) < min_bars_needed or pd.isna(data["MA"].iloc[-2]): return None, None latest, prev = data.iloc[-1], data.iloc[-2] current_close, current_ma = float(latest["Close"]), float(latest["MA"]) previous_close, previous_ma = float(prev["Close"]), float(prev["MA"]) current_ema200 = float(latest["EMA200"]) current_volume, current_vol_avg = float(latest["Volume"]), float(latest["VolAvg"]) volume_ok = ( not pd.isna(current_vol_avg) and current_vol_avg > 0 and current_volume >= current_vol_avg * VOLUME_SURGE_MULTIPLIER ) if previous_close < previous_ma and current_close > current_ma: if current_close > current_ema200 and volume_ok: return "bullish", current_close elif previous_close > previous_ma and current_close < current_ma: if current_close < current_ema200 and volume_ok: return "bearish", current_close return None, current_close def already_holds_option_on(underlying): """True if we currently hold ANY open option position on this underlying (call or put) — prevents stacking multiple positions on the same stock.""" try: positions = api.list_positions() for p in positions: if getattr(p, "asset_class", "") == "us_option" and p.symbol.startswith(underlying): return True except Exception as e: print(f"⚠️ Could not check existing option positions: {e}") return False # ============================================================================== # STEP 6: SCAN JOB — open new option positions on fresh signals # ============================================================================== def run_options_scan(): print(f"⏰ OPTIONS SCAN: {datetime.datetime.now()}") for ticker in WATCHLIST: try: signal, current_price = get_signal(ticker) if signal is None: print(f"-> {ticker}: no qualifying signal.") continue if already_holds_option_on(ticker): print(f"-> {ticker}: bullish/bearish signal but already holding an option on this underlying — skipping.") continue option_type = "call" if signal == "bullish" else "put" contracts = get_option_contracts(ticker, option_type) contract = choose_atm_contract(contracts, current_price) if not contract: print(f"-> {ticker}: no suitable {option_type} contract found in the {TARGET_EXPIRATION_MIN_DAYS}-{TARGET_EXPIRATION_MAX_DAYS} day window — skipping.") continue option_symbol = contract["symbol"] mid_price = get_option_mid_price(option_symbol) if mid_price is None or mid_price <= 0: print(f"-> {ticker}: couldn't get a live quote for {option_symbol} — skipping.") continue contract_cost = mid_price * 100 # each contract = 100 shares of exposure if contract_cost > MAX_CONTRACT_COST_CAP: print(f"-> {ticker}: 1 contract of {option_symbol} costs ${contract_cost:.2f}, above the ${MAX_CONTRACT_COST_CAP:.0f} cap — skipping.") continue qty = max(1, int(BUDGET_PER_TRADE // contract_cost)) print(f"-> {ticker}: {signal} signal, buying {qty}x {option_symbol} @ ~${mid_price:.2f}/contract (${contract_cost * qty:.2f} total).") submit_option_order_and_notify( option_symbol, qty, "buy", f"{signal.title()} 5-min MA crossover on {ticker} (200 EMA trend + volume confirmed). " f"Strike ${contract['strike_price']}, exp {contract['expiration_date']}." ) time.sleep(1.5) except Exception as e: print(f"❌ Error scanning {ticker}: {e}") send_alert(f"🚨 OPTIONS SCANNER ERROR on {ticker}: {e}") # ============================================================================== # STEP 7: MANAGE JOB — stop-loss / take-profit / end-of-day flatten (runs every minute) # ============================================================================== def is_near_market_close(minutes_before): try: clock = api.get_clock() if not clock.is_open: return False close_time = clock.next_close now = clock.timestamp minutes_remaining = (close_time - now).total_seconds() / 60 return minutes_remaining <= minutes_before except Exception as e: print(f"⚠️ Could not check market close time: {e}") return False def run_options_manage(): print(f"⏱️ OPTIONS MANAGE: {datetime.datetime.now()}") try: positions = api.list_positions() except Exception as e: print(f"⚠️ Could not fetch positions: {e}") return option_positions = [p for p in positions if getattr(p, "asset_class", "") == "us_option"] if not option_positions: return eod = is_near_market_close(EOD_FLATTEN_MINUTES_BEFORE_CLOSE) for pos in option_positions: symbol = pos.symbol qty = float(pos.qty) entry_price = float(pos.avg_entry_price) current_price = float(pos.current_price) if pos.current_price else None if current_price is None or entry_price <= 0: continue pl_pct = (current_price - entry_price) / entry_price if eod: send_alert(f"🕐 [END OF DAY FLATTEN] Closing {symbol} (this bot never holds overnight). P/L: {pl_pct*100:+.2f}%") submit_option_order_and_notify(symbol, abs(qty), "sell", "End-of-day flatten — day-trading bot, no overnight holds.") continue if pl_pct <= -OPTION_STOP_LOSS_PCT: send_alert(f"🔻 [OPTION STOP-LOSS] {symbol} down {pl_pct*100:.2f}% (limit -{OPTION_STOP_LOSS_PCT*100:.0f}%). Closing.") submit_option_order_and_notify(symbol, abs(qty), "sell", f"Premium stop-loss hit at {pl_pct*100:.2f}%.") continue if pl_pct >= OPTION_TAKE_PROFIT_PCT: send_alert(f"🎯 [OPTION TAKE-PROFIT] {symbol} up {pl_pct*100:.2f}% (target +{OPTION_TAKE_PROFIT_PCT*100:.0f}%). Closing.") submit_option_order_and_notify(symbol, abs(qty), "sell", f"Premium take-profit hit at +{pl_pct*100:.2f}%.") continue print(f"-> {symbol}: holding, P/L {pl_pct*100:+.2f}% (stop -{OPTION_STOP_LOSS_PCT*100:.0f}%, target +{OPTION_TAKE_PROFIT_PCT*100:.0f}%).") # ============================================================================== # STEP 8: ENTRY POINT # ============================================================================== def is_market_hours(): try: clock = api.get_clock() return bool(clock.is_open) except Exception as e: print(f"⚠️ Could not reach Alpaca clock: {e}") eastern = zoneinfo.ZoneInfo("America/New_York") now = datetime.datetime.now(eastern) if now.weekday() >= 5: return False return now.replace(hour=9, minute=30, second=0) <= now <= now.replace(hour=16, minute=0, second=0) if __name__ == "__main__": job = sys.argv[1] if len(sys.argv) > 1 else "scan" if job == "scan": if is_market_hours(): run_options_scan() else: print("Outside market hours — skipping options scan.") elif job == "manage": if is_market_hours(): run_options_manage() else: print("Outside market hours — skipping options manage.") else: print(f"Unknown job '{job}'. Use one of: scan, manage") sys.exit(1)
